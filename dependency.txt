# Dependencies and compilation instructions

# Install required packages
sudo apt update
sudo apt install build-essential
sudo apt install libgtk-3-dev
sudo apt install libpthread-stubs0-dev

# ===== VOICE RECOGNITION DEPENDENCIES (ENHANCED) =====
# Core audio and Python dependencies
sudo apt install python3 python3-pip sox libsox-fmt-all pulseaudio-utils alsa-utils portaudio19-dev python3-dev

# Python speech recognition libraries
pip3 install --user SpeechRecognition pyaudio pocketsphinx

# Quick setup (automated)
./setup_voice.sh

# Compile command:
gcc main.c shell_functions.c callbacks.c utils.c auto_suggest.c voice_recognition.c kernel_features.c -o custom_shell `pkg-config --cflags --libs gtk+-3.0` -lm -lpthread

# Run:
./custom_shell

# Quick Demo:
./demo_voice.sh

# ===== VOICE RECOGNITION FEATURES =====
# ✅ REAL voice recognition implemented (no longer simulated!)
# ✅ Multiple recognition engines:
#    1. Google Speech API (online, highest accuracy)
#    2. PocketSphinx (offline fallback)
#    3. Smart command cycling (ultimate fallback)

# ===== SUPPORTED VOICE COMMANDS =====
# Natural Language Commands:
# • "list files" / "show me files" → ls -la
# • "show current directory" / "where am i" → pwd
# • "calculate 2 plus 2" / "compute" → calc 2+2
# • "clear screen" / "clear terminal" → clear
# • "change directory home" → cd ~
# • "create folder test" → mkdir test
# • "show processes" → ps aux
# • "disk usage" → df -h
# • "memory usage" → free -h
# • "network status" → ip addr
# • "system info" → uname -a

# Direct Commands (also supported):
# • "ls", "pwd", "date", "clear", "help", etc.
